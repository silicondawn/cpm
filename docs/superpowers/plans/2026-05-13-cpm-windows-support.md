# cpm Windows Support Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Windows (PowerShell 7+) support to `cpm` without breaking existing macOS/Linux behavior, using Go build tags to split platform-specific implementations.

**Architecture:** Function-level platform abstraction via build-tagged `_unix.go` / `_windows.go` file pairs. No interfaces. Junction-first directory sharing on Windows with symlink fallback. PowerShell prompt-function hook for auto-switch. `.cmd` + `.ps1` wrapper pair per profile.

**Tech Stack:** Go 1.26, cobra, BurntSushi/toml. Windows-only support is plain `os/exec` (no `golang.org/x/sys/windows` dependency in v1). Tests via standard `testing` + `go test`. CI on GitHub Actions matrix (ubuntu-latest, macos-latest, windows-latest). Release via goreleaser with new scoop publisher.

```go
// internal/wrapper.go (post-refactor)

package internal

const marker = "# Generated by cpm (Claude Profile Manager)"

// WrapperFiles is what GenerateWrapper returns.
// Unix: one entry, no sidecar (Main="claude-<name>", MainContent="#!/usr/bin/env bash...").
// Windows: two entries (Main="claude-<name>.cmd", Sidecar="claude-<name>.ps1").
type WrapperFiles struct {
    Main         string // base filename only, e.g. "claude-personal" or "claude-personal.cmd"
    MainContent  string
    Sidecar      string // empty on Unix
    SidecarContent string
    // Mode for written files. Honored on Unix (0o755). Set but ignored on Windows.
    Mode uint32
}

// GenerateWrapper is platform-split.
// Unix: wrapper_unix.go
// Windows: wrapper_windows.go
func GenerateWrapper(name, profileDir string, profile *Profile) WrapperFiles

// InstallWrapper writes the files into binDir. Cross-platform (in wrapper.go).
// It returns the path of the main file (used by caller for log output).
func InstallWrapper(binDir string, files WrapperFiles) (mainPath string, err error)

// wrapperFilenames lists all filenames a given profile owns in binDir.
// Used by CleanupStaleScripts and doctor.
// Unix: paths_unix.go (returns ["claude-<name>"])
// Windows: paths_windows.go (returns ["claude-<name>.cmd", "claude-<name>.ps1"])
func wrapperFilenames(profileName string) []string

// CleanupStaleScripts walks binDir and removes any file containing the marker
// that does not belong to an active profile. Cross-platform.
func CleanupStaleScripts(binDir string, activeProfileNames map[string]bool)
```

```go
// resolveLinkTarget reads the link target and normalizes it
```

```go
//go:build windows

package internal

import (
    "fmt"
    "os"
    "os/exec"
    "path/filepath"
    "strings"
)

// linkShared creates dst as a directory link pointing to src.
// Strategy: junction for same-volume directories (no privilege needed);
// symlink fallback for cross-volume (requires Developer Mode).
func linkShared(src, dst string) error {
    src = filepath.Clean(src)
    dst = filepath.Clean(dst)

    if sameVolume(src, dst) {
        if err := mklinkJunction(dst, src); err == nil {
            return nil
        }
        // junction failed (extremely rare); fall through to symlink
    }

    if err := os.Symlink(src, dst); err == nil {
        return nil
    } else if isPrivilegeNotHeldError(err) {
        return fmt.Errorf(
            "cannot create symlink to %s — Windows requires Developer Mode for cross-volume symlinks.\n"+
                "Enable: Settings → Privacy & Security → For developers → Developer Mode = On\n"+
                "Then re-run 'cpm install'.", src)
    } else {
        return fmt.Errorf("cannot create symlink to %s: %w", src, err)
    }
}

// resolveLinkTarget reads the link target and normalizes Windows junction
// prefixes (\??\ or \\?\) out, returning a path comparable to the original src.
func resolveLinkTarget(path string) (string, error) {
    target, err := os.Readlink(path)
    if err != nil {
        return "", err
    }
    // Junctions on Windows surface a \??\ prefix; symlinks may surface \\?\
    target = strings.TrimPrefix(target, `\??\`)
    target = strings.TrimPrefix(target, `\\?\`)
    return filepath.Clean(target), nil
}

func sameVolume(a, b string) bool {
    return strings.EqualFold(filepath.VolumeName(a), filepath.VolumeName(b))
}

func mklinkJunction(dst, src string) error {
    cmd := exec.Command("cmd", "/c", "mklink", "/J", dst, src)
    out, err := cmd.CombinedOutput()
    if err != nil {
        return fmt.Errorf("mklink /J: %s: %w", strings.TrimSpace(string(out)), err)
    }
    return nil
}

func isPrivilegeNotHeldError(err error) bool {
    return strings.Contains(err.Error(), "A required privilege is not held by the client")
}
```

```go
package internal

import (
    "fmt"
    "os"
    "path/filepath"
    "strings"
)

const marker = "# Generated by cpm (Claude Profile Manager)"

// Subcommands that pass through unmodified to claude.
const subcommands = "mcp|auth|doctor|install|setup-token|update|upgrade|agents|auto-mode|plugin|plugins"

// WrapperFiles describes the file(s) GenerateWrapper produces.
// Unix: Main + MainContent. Sidecar empty.
// Windows: Main = "claude-<name>.cmd", Sidecar = "claude-<name>.ps1".
type WrapperFiles struct {
    Main           string // base filename (no directory)
    MainContent    string
    Sidecar        string // base filename; empty if no sidecar
    SidecarContent string
    Mode           uint32 // file mode (e.g. 0o755 on Unix; ignored on Windows)
}

// InstallWrapper writes the wrapper file(s) into binDir.
// Returns the path of the main file (for the caller's log output).
// Cross-platform.
func InstallWrapper(binDir string, files WrapperFiles) (string, error) {
    if err := os.MkdirAll(binDir, 0o755); err != nil {
        return "", err
    }

    mainPath := filepath.Join(binDir, files.Main)
    if err := writeIfChanged(mainPath, files.MainContent, files.Mode); err != nil {
        return "", err
    }

    if files.Sidecar != "" {
        sidecarPath := filepath.Join(binDir, files.Sidecar)
        if err := writeIfChanged(sidecarPath, files.SidecarContent, files.Mode); err != nil {
            return "", err
        }
    }

    return mainPath, nil
}

func writeIfChanged(path, content string, mode uint32) error {
    if existing, err := os.ReadFile(path); err == nil && string(existing) == content {
        return nil
    }
    if err := os.WriteFile(path, []byte(content), os.FileMode(mode)); err != nil {
        return err
    }
    fmt.Printf("  installed %s\n", path)
    return nil
}

// CleanupStaleScripts walks binDir and removes any file containing the
// cpm-generated marker that does not belong to an active profile.
func CleanupStaleScripts(binDir string, activeProfileNames map[string]bool) {
    entries, err := os.ReadDir(binDir)
    if err != nil {
        return
    }

    // Build the active filename set from the active profile names.
    activeFiles := make(map[string]bool, len(activeProfileNames)*2)
    for prof := range activeProfileNames {
```

```go
    b.WriteString(":: " + marker + "\n")

// psQuote single-quotes a string for embedding in PowerShell, doubling embedded single quotes.
func psQuote(s string) string {
    return "'" + strings.ReplaceAll(s, "'", "''") + "'"
}

func generateWindowsPS1(name, profileDir string, profile *Profile) string {

    b.WriteString("# " + marker + "\n")

    b.WriteString("# Unset inherited CLAUDE_*/ANTHROPIC_* env vars\n")
```

```go
}
if [ -n "$ZSH_VERSION" ]; then
```

```go
func ExecClaude(claudePath string, args []string, env []string) error {
```

```go
//go:build windows

package internal

import (
    "fmt"
    "os"
)

func replaceBinary(tmpPath, targetPath string) error {
    oldPath := targetPath + ".old"
    _ = os.Remove(oldPath) // clean stragglers from previous upgrade

    if _, err := os.Stat(targetPath); err == nil {
        if err := os.Rename(targetPath, oldPath); err != nil {
            return fmt.Errorf("rename current binary aside: %w", err)
        }
    }

    if err := os.Rename(tmpPath, targetPath); err != nil {
        _ = os.Rename(oldPath, targetPath) // best-effort rollback
        return fmt.Errorf("install new binary: %w", err)
    }
    return nil
}

// CleanupOldBinary silently removes targetPath+".old" if present.
// Called once at startup so the next upgrade can rename to .old again.
func CleanupOldBinary(targetPath string) {
    _ = os.Remove(targetPath + ".old")
}
```
